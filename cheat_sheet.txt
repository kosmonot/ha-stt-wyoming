Local faster-whisper (GPU) STT service for Home Assistant, run via NSSM on Win11.

.\venv\Scripts\activate

START WYOMING SERVER

cd C:\Users\john\Projects\ha-stt-wyoming
.\venv\Scripts\activate

mkdir .\data -Force

python -m wyoming_faster_whisper `
  --uri "tcp://0.0.0.0:10300" `
  --language en `
  --model medium `
  --device cuda `
  --data-dir ".\data" `
  --download-dir ".\data"


**********
nssm install HA_STT C:\Users\john\Projects\ha-stt-wyoming\run_stt.cmd
nssm set HA_STT AppDirectory C:\Users\john\Projects\ha-stt-wyoming
nssm set HA_STT Start SERVICE_AUTO_START

*****************

 Stop-Process -Name python -Force
 PS C:\Users\john\Projects\ha-stt-wyoming\logs> Get-Content .\stderr.log -Tail 50 -Wait
